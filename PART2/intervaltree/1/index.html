    rect area

    Given a list of axis-aligned rectangles. Each rectanges[i] = [x1, y1, x2, y2], where [x1, y1] are the coordinates of the bottom-left corner, and (x2, y2) are the coordinates of the top-right corner of the $i_{th}$ rectangle.

    Find the total area coverd by all rectangles in the plane. Since the answer may be too large, return it modulo 10 ^ 9 + 7.

    Examples: 

    Input: [[0,0,2,2],[1,0,2,3],[1,0,3,1]]
Output: 6

    Input: [[0,0,1000000000,1000000000]]
Output: 49
Explanation: The answer is 10^18 modulo (10^9 + 7), which is (10^9)^2 = (-7)^2 = 49.
    Note: 
    - 1 <= rectangles.length <= 200
- rectangles[i].length = 4
- 0 <= rectangles[i][j] <= 10^9
- the total area coverd by all rectangles will never exceed 2^63-1 and thus will 
  fit in a 64-bit signed integer.

    解题步骤

    (1). Sweep Line 算法

     视 $x$ 轴被矩形的竖边划分若干个区域，假设有一条平行于 $y$ 轴的扫描线从最左侧开始扫描，目的是每扫描到一个区域就计算出有效面积，最后把所有值累加得到结果。
     可以观察到，每当扫描线位于一个区域中时，被扫描线穿过的矩形左侧边一定被扫描过，且右侧边一定未被扫描，因此每个区域的面积 = 被穿过的矩形的左边在 $y$ 轴投影的并集 × 区域的宽。

    Steps of Sweep-Line:
把竖边排序，从左到右依次扫描。
    1. line = 当前扫到的边
    2. area += 两边之差 × 另一边投影的并集长度
    3. if line 是左侧边：line 的每个单位长度的覆盖次数加一
    4. else:            line 的每个单位长度的覆盖次数减一
    因此，需要一个数据结构动态维护区间的覆盖信息，线段树是一个很好的选择。

    (2). 线段树

    此题对于线段树的限制有：

    • 区间跨度大且分散，需要离散化；
    • 需要区间修改，且一次加一一定对应一次减一；
    • 每个结点需要维护区间内被覆盖单位的并集长度；
    • 只有根结点需要被查询。
    因此，每个结点维护两个属性：并集长度和修改的标记。这里使用永久化标记的方法实现区间修改，但由于要维护并集长度，需要额外注意：
    • 只有在当前结点未被标记的情况下，遍历它的左右子树之后，才需要更新它的并集长度属性；
    • 当修改一段目的区间的标记值后，需要判断它是否刚好被标记一次 或 标记刚好被删除干净，依此判断如何更新其并集长度属性；如果是后者，还要注意是否为叶结点。

    #include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <cstdlib>
#include <cstdio>

using namespace std;

const int LEFT = 1, RIGHT = 2;

struct Line {

    int direction, y1, y2, x;

    Line(int direction, int y1, int y2, int x) {
        this->direction = direction;
        this->y1 = y1;
        this->y2 = y2;
        this->x = x;
    }

    bool operator < (Line line) {
        if (this->x < line.x) 
            return true;
        return false;
    }

};

struct SegTree {

    int *tree;
    unsigned char *addsum;
    vector<int> ymap;

    SegTree (set<int> ycoords) {
        ymap = vector<int>(ycoords.begin(), ycoords.end());
        tree = (int *) malloc(sizeof(int) * ymap.size() * 4);
        addsum = (unsigned char *) malloc(sizeof(char) * ymap.size() * 4);
        build(0, 0, ymap.size());
    }

    void build(int index, int l, int r) {
        tree[index] = addsum[index] = 0;
        if (l == r - 1) 
            return;
        int mid = l + (r - l) / 2;
        build(index * 2 + 1, l, mid);
        build(index * 2 + 2, mid, r);
    }

    int query_tree() {
        return tree[0];
    }

    void modify_tree(Line line) {
        auto beg = ymap.begin(), end = ymap.end();
        auto it1 = lower_bound(beg, end, line.y1);
        auto it2 = lower_bound(beg, end, line.y2);
        int leftval = (line.direction == LEFT) ? (1) : (-1);
        modify(0, 0, ymap.size(), (int)(it1 - beg), (int)(it2 - beg), leftval);
    }

    void modify(int index, int lc, int rc, int ls, int rs, int val) {
        if (lc >= ls && rc <= rs) {
            addsum[index] += val;
            if (addsum[index] == 1) {
                tree[index] = ymap[rc] - ymap[lc];
            } else if (addsum[index] == 0) {
                if (lc != rc - 1)
                    tree[index] = tree[2 * index + 1] + tree[2 * index + 2];
                else
                    tree[index] = 0;
            }
            return;
        }
        int mid = lc + (rc - lc) / 2;
        if (ls < mid)
            modify(index * 2 + 1, lc, mid, ls, rs, val);
        if (rs > mid)
            modify(index * 2 + 2, mid, rc, ls, rs, val);
        if (addsum[index] == 0)
            tree[index] = tree[2 * index + 1] + tree[2 * index + 2];
    }
};

class Solution {
    public:
        int rectangleArea(vector<vector<int>>& rectangles) {
            int size = rectangles.size() * 2;
            set<int> ylinenum;
            if (size == 0) 
                return 0;
            vector<Line> lines;
            lines.reserve(size);
            for (vector<int> &rect : rectangles) {
                lines.emplace_back(LEFT, rect[1], rect[3], rect[0]);
                lines.emplace_back(RIGHT, rect[1], rect[3], rect[2]);
                ylinenum.insert(rect[1]);
                ylinenum.insert(rect[3]);
            }
            sort(lines.begin(), lines.end());
            long long area = 0;
            SegTree tree(ylinenum);
            tree.modify_tree(lines[0]);
            int last = lines[0].x;
            for (auto it = lines.begin() + 1; it != lines.end(); ++it) {
                area += (it->x - last) * (long long)(tree.query_tree());
                tree.modify_tree(*it);
                last = it->x;
            }
            return area % 1000000007;
        }
};
