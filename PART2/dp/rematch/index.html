    expression matching

    regular expression matching

    Given an input string (s) and a pattern (p), implement regular expression matching with support for '.' and '*'.

    Examples: 

     Input:  s = "aab", p = "c*a*b"
 Output: true

     Input:  s = "ab", p = ".*"
 Output: true
    Note: 
    - '.' Matches any single character.
- '*' Matches zero or more of the preceding element.
- s could be empty and contains only lowercase letters a-z.
- p could be empty and contains only lowercase letters a-z, and characters like . or *.
    (Source: LeetCode 10)

    $\small \mathrm{dp}[i][j]: \mathsf{if\ source[0...i]\ matches\ with\ pattern[0...j]}$
    $\small \mathrm{dp}[i][j]=(\mathrm{dp}[i][j-1]\ $&&$\small\ \mathit{matches}(\epsilon,\mathrm{p}[j-1]))\ $||$\small\ (\mathrm{dp}[i-1][j-1]\ $&&$\small\ \mathit{matches}(\mathrm{s}[i...i],\mathrm{p}[j-1]))\ $|| ...

    为方便处理 $\small \mathrm{p}[0...]$ 被空匹配的情况，在 $\small\mathrm{s}$ 和 $\small\mathrm{p}$ 之前添加了一个相同的字符。

    Code
    bool isMatch(string s, string p) {
    p = "a" + p;
    s = "a" + s;
    vector<int> dest = generateIndex(p);  
    int M = s.size(), N = dest.size() - 1;
    bool dp[M][N]; 
    string t = p.substr(0, dest[1] - dest[0]);
    for (int i = 0; i < M; ++i) 
        dp[i][0] = singleMatch(s.substr(0, i + 1), t);

    for (int j = 1; j < N; ++j) {        
        string pi = p.substr(dest[j], dest[j + 1] - dest[j]);
        for (int i = 0; i < M; ++i) {
            bool t = false;
            for (int k = 0; k <= i; ++k) {
                t = t || (dp[i - k][j - 1] && singleMatch(s.substr(i + 1 - k, k), pi));
                if (t) break;
            }
            dp[i][j] = t;
        }
    }

    return dp[M - 1][N - 1];
}

vector<int> generateIndex(string& p) {
    vector<int> ret;
    int i;
    for (i = 0; i < p.size(); ++i) {
        if (p[i] == '*')
            continue;
        ret.push_back(i);
    }
    ret.push_back(i);
    return ret;
}

bool singleMatch(const string& src, const string& pat) {
    if (pat[0] == '.') {
        if (pat.size() == 1)
            return src.size() == 1;
        return true;
    } 
    if (pat.size() == 1)
        return src.size() == 1 && src[0] == pat[0];
    for (auto& c : src)
        if (c != pat[0])
            return false;
    return true;
}



    wildcard matching (通配符匹配)

    Given an input string (s) and a pattern (p), implement wildcard pattern matching with support for '?' and ''.

    Examples: 

     Input:  s = "adceb", p = "*a*b"
 Output: true

     Input:  s = "", p = "*"
 Output*: true
    Note: 
    - '?' Matches any single character.
- '*' Matches any sequence of characters (including the empty sequence).
- s could be empty and contains only lowercase letters a-z.
- p could be empty and contains only lowercase letters a-z, and characters like ? or *.
    (Source: LeetCode 44)

    类似于上面一题，此题也可以通过 DP 解决，思路基本一样，但这里需要优化一下。

    Code: DP (Only Doing Time Optimization)
    bool isMatch(string s, string p) {
    p = "a" + p;
    s = "a" + s; 
    int M = s.size(), N = p.size();
    bool dp[M][N]; 
    bool tagp[M], tagc[M];
    bool *tagpre = tagp, *tagcur = tagc;
    dp[0][0] = true;
    tagpre[0] = true;
    for (int i = 1; i < M; ++i) {
        dp[i][0] = false;
        tagpre[i] = true;
    }

    for (int j = 1; j < N; ++j) {        
        bool havetrue = false;
        for (int i = 0; i < M; ++i) {
            bool t = false;
            if (p[j] == '*') {
                t = tagpre[i];
            } else if (i != 0) {
                t = dp[i - 1][j - 1] && singleMatch(s.substr(i, 1), p[j]);
            }
            dp[i][j] = t;
            if (t) havetrue = true;
            tagcur[i] = havetrue;
        }
        bool *tmp = tagcur;
        tagpre = tagcur;
        tagcur = tmp;
    }
    return dp[M - 1][N - 1];
}

bool singleMatch(const string& src, char& pat) {
    if (pat == '*')
        return true;
    if (pat == '?') 
        return src.size() == 1;
    return src.size() == 1 && src[0] == pat;     
}

     但此题可用贪心解决，两种方法的效率的差别体现在对 $*$ 的处理上。
     主体思路是，从左向右扫描 $\mathrm{s}$ 和 $\mathrm{p}$，当遇到 $\mathrm{p}$ 中的 $*$ 时，记录它的位置，暂令其匹配 $\mathrm{s}$ 中长度为 $x$ 的串；若后面的匹配过程成功，则返回 $true$；否则，回溯到记录的位置，再尝试令 $*$ 匹配长为 $x+1$ 的串，如此循环，直到成功或失败。
     贪心性质的体现在于，回溯只需要回溯到距 $\mathrm{p}$ 中指针最近的一个 $*$ 即可。

    结论：设 $\mathrm{s}=s_1s_2s_3, \mathrm{p}=p_1*s_2*p_3$，其中 $s_{ij}\in\lbrace a...z\rbrace$，$p_{ij}\in\lbrace a...z\rbrace\cup\lbrace*, ?\rbrace$，若 $\mathrm{s}$ 与 $\mathrm{p}$ 匹
       配且 $s_1s_2$ 与 $p_1*s_2$ 匹配，那么 $s_3$ 一定与 $*\ p_3$ 匹配。

    说明：假设 $s_3$ 不与 $*p_3$ 匹配，那么回溯到 $s_2$ 之前的那个 $*$，令其匹配更多的元素，同时还要在 $\mathrm{s}$ 中的一个新的位置匹配 $s_2$ (假设此位置存在)，$\mathrm{s}$ 剩下的部分记为 $s_3[i...]$。由 $s_3$ 不与 $*p_3$ 匹配，则左侧截去一部分后也不匹配 (由于 $p_3$ 前面的 $*$ 可吸收一切字符)，则 $\mathrm{s}$ 与 $\mathrm{p}$ 不匹配，矛盾产生。

    Code: Greedy
    // ver1
// the behavior equals to ver2
bool isMatch(string s, string p) {
    s += "-", p += "-";
    int i = 0, j = 0, pi = -1, pj = -1;
    while (i < s.size()) {
        if (s[i] == p[j] || p[j] == '?') {
            ++i, ++j;
        } else if (p[j]== '*') {
            pi = i;
            pj = j++;
        } else if (pi >= 0) {
            i = ++pi;
            j = pj + 1;           
        } else return false;
    }
    return j == p.size();
}

// ver2
// i j 同时扫描完：成功退出
// i 先扫描完，则需要判断 j 剩下的部分是否都是 *
// j 先扫描完，失配，进入 case3，回溯到上一个 * 的位置 (*p3 <--> s3)
bool isMatch(string s, string p) {
    int i = 0, j = 0, pi = -1, pj = -1;
    while (i < s.size()) {
        if ((s[i] == p[j] || p[j] == '?') && j < p.size()) { // case1
            ++i, ++j;
        } else if (j < p.size() && p[j] == '*') { // case2
            pi = i;
            pj = j++;
        } else if (pi >= 0) { // case3
            i = ++pi;
            j = pj + 1;
        } else return false; // case4
    }
    while (j < p.size() && p[j] == '*') ++j;
    return j == p.size();
}

    当 $i$ 先越界，设 $\mathrm{p}$ 剩下的串为 $r$，仍沿用结论中的符号，说明 $s_3$ 和 $*p_3$(可以假设 $*$ 存在且 $p_3$ 不为空) 匹配。由于 $p_3$ 中在此处不包含 $*$，因此 $\mathrm{s}$ 与 $\mathrm{p}$ 匹配等价于 $r$ 中元素全是 $*$。
