    void traverse(Node *root) {
    if (root != NULL) {
        printf("%c ", root->name);
        traverse(root->left);       
        traverse(root->right);
    }
}
     通过递归遍历一棵二叉树，递归的过程相当于沿着树的外沿周游一圈。对每一个结点来说，按照向左右递归划分出了三种访问时机，在每个时机访问对应了根和左右子树之间的一个相对访问次序，分别为前序、中序、后序。
     以下遍历算法所需时间均为 $\Theta (n)$。

    ① 非递归先序、中序遍历
     可以通过辅助栈，入栈模拟结点的先序时机，出栈模拟中序时机，迭代终止的条件为栈空且当前的结点指针也为空。

    void traverse(Node *root) {
    Node *stack[MAX];
    int top = -1;
    do {
        while (root != NULL) {
            stack[++top] = root; // STACK-PUSH: Pre
            printf("%c ", root->name);
            root = root->left;
        }
        root = stack[top--];     // STACK-POP: In
        root = root->right;     
    } while (top != -1 || root != NULL);
}

    ② 非递归后序遍历
     入栈模拟先序时机，出栈模拟后序时机。
     1. 如果当前指针非空，进栈、左移，模拟左递归；
     2. 如果当前指针为空，记录上一个访问的结点 last，判断其是否为栈顶结点的右孩子：如果不是，说明栈顶结点的右子树尚未被访问，当前指针设为右孩子并重复 1；如果是，说明右子树访问完成，退栈、模拟后序时机，并继续重复判断下一个栈顶结点。

    void postorderTraverse(Node *root) {
    Node *stack[MAX];
    int top = -1;
    do {
        while (root != NULL) {
            stack[++top] = root;
            root = root->left;
        }       
        Node *last = NULL;
        while (top != -1) {
            Node *t = stack[top];
            if (last == t->right) {
                printf("%c ", t->name);
                last = t;
                t = stack[--top];
            } else {
                root = t->right;
                break;
            }
        }
    } while (top != -1);
}
     第二个迭代需要判断栈非空。

    void levelTraverse(Node *root) {
    Node *queue[MAX];
    int head = -1;
    int tail = -1;
    queue[++tail] = root;
    while (head != tail) {
        Node *t = queue[++head];
        if (t->left) {
            queue[++tail] = t->left;
        }
        if (t->right) {
            queue[++tail] = t->right;
        }
    }
}
