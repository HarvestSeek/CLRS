    /* build a interval [l, r) */
void build(int idx, int l, int r) {
    if (l + 1 == r) {
        nodes[idx] = attr[l];
        return;
    }
    int mid = l + (r - l) / 2;
    build(idx * 2 + 1, l, mid);
    build(idx * 2 + 2, mid, r);
    nodes[idx] = nodes[idx * 2 + 1] + nodes[idx * 2 + 2];
}

int query(int idx, int lc, int rc, int ls, int rs) {
    if (lc >= ls && rc <= rs) {
        return nodes[idx];
    }
    int mid = lc + (rc - lc) / 2, res = 0;
    if (ls < mid)

        res += query(idx * 2 + 2, mid, rc, ls, rs);

}

    void modify(int idx, int lc, int rc, int ls, int rs, int val) {
    if (lc >= ls && rc <= rs) {
        addsum[idx] += val;
        return;
    }
    nodes[idx] += (min(rs, rc) - max(ls, lc)) * val;
    int mid = lc + (rc - lc) / 2;
    if (ls < mid)
        modify(idx * 2 + 1, lc, mid, ls, rs, val);
    if (rs > mid)
        modify(idx * 2 + 2, mid, rc, ls, rs, val);
}

int query(int idx, int lc, int rc, int ls, int rs) {
    if (lc >= ls && rc <= rs) {
        return nodes[idx] + (rc - lc) * addsum[idx];
    }
    int mid = lc + (rc - lc) / 2;
    int res = (min(rs, rc) - max(ls, lc)) * addsum[idx];
    if (ls < mid)
        res += query(idx * 2 + 1, lc, mid, ls, rs);
    if (rs > mid)
        res += query(idx * 2 + 2, mid, rc, ls, rs);
    return res;
}
